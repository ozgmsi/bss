60 // Since a timeval has to separate parts, doing
61 // arithmetic on them may produce abnormal part values.
62 void normalizeTimeval(timeval& x)
63 {
64  while (x.tv_sec < 0) { // "borrow" from usec
65  x.tv_sec += 1;
66  x.tv_usec -= R_USAGE_QUOTUM;
67  }
68  while (x.tv_usec < 0) { // "borrow" from sec
69  x.tv_sec -= 1;
70  x.tv_usec += R_USAGE_QUOTUM;
71  }
72  while (x.tv_usec >= R_USAGE_QUOTUM) { // "carry" usec to sec
73  x.tv_sec += 1;

75  }
