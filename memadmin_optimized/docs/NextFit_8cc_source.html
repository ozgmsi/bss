122 bool NextFit::reclaim()
123 {
124  bool changed = false; // did we change anything ?
125 
126  // Sort resource map by area address
127  areas.sort( Area::orderByAddress() ); // WARNING: expensive N*log(N) operation !
128 
129  // Search thru all available areas for matches
130  ALiterator i = areas.begin();
131  Area *ap = *i; // the current candidate ...
132  for (++i ; i != areas.end() ;) {
133  Area *bp = *i; // ... match ap with ...
134  if (bp->getBase() == (ap->getBase() + ap->getSize()))
135  {
136  // oke, bp matches ap ...

138  ap->join(bp); // append area bp to ap (and destroy bp)

146  }
