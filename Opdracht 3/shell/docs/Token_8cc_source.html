94 // Note: The "if (!is.get(c)) ..." construct used below
95 // is needed to handle "unexpected EOF" situations.
96 Token *Token::nextToken(istream& is)
97 {
98  string s;
99  char c = -1;
100 
101  // Worth the trouble doing anything ?
102  if (!is)
103  return new Token(Token::END);
104 
105  // Get one character (if possible)
106  if (!is.get(c))
107  return new Token(Token::END);
108 
109  // Skip any white space ...
110  while (isBlank(c)) {
111  if (!is.get(c))
112  return new Token(Token::END);
113  }
114 
115  // What have we got now ?
116 
117  switch(c) { // First look for special cases
118 
119  case '\n': case '\r':
120  return new Token(Token::EOL);
121 
122  case ';': // Can be either ; or ;;
123  if (Token::level >= Token::SH) { // iff sh-script: could be ;;
124  if (!is.get(c)) // lookahead
125  return new Token(Token::SEQUENCE);
126  if (c == ';')
127  return new Token(Token::BREAK);
128  is.putback(c); // undo lookahead
129  }
130  return new Token(Token::SEQUENCE);
131 
132  case '&': // Can be either & or &&
133  if (Token::level > Token::BASIC) { // iff advanced: could be &&
134  if (!is.get(c)) // lookahead
135  return new Token(Token::BACKGROUND);
136  if (c == '&')
137  return new Token(Token::AND);
138  is.putback(c); // undo lookahead
139  }
140  return new Token(Token::BACKGROUND);
141 
142  case '|': // Can be either | or ||
143  if (Token::level > Token::BASIC) { // if advanced: could be ||
144  if (!is.get(c)) // lookahead
145  return new Token(Token::PIPE);
146  if (c == '|')
147  return new Token(Token::OR);
148  is.putback(c); // undo lookahead
149  }
150  return new Token(Token::PIPE);
151 
152  case '>': // Can be either > or >>
153  if (!is.get(c)) // lookahead
154  return new Token(Token::OUTPUT);
155  if (c == '>')
156  return new Token(Token::APPEND);
157  is.putback(c); // undo lookahead
158  return new Token(Token::OUTPUT);
159 
160  case '<': // Can be either < or <<
161  if (Token::level >= Token::SH) { // if sh-scripts: could be <<
162  if (!is.get(c)) // lookahead
163  return new Token(Token::INPUT);
164  if (c == '<')
165  return new Token(Token::HERE);
166  is.putback(c); // undo lookahead
167  }
168  return new Token(Token::INPUT);
169 
170  case '(': // subshell begin
171  if (Token::level == Token::BASIC) // must at least be advanced level
172  break; // treat like ordinary char
173  return new Token(Token::SUBS);
174 
175  case ')': // subshell end
176  if (Token::level == Token::BASIC) // must at least be advanced level
177  break; // treat like ordinary char
178  return new Token(Token::ENDS);
179 
180  case '$': // variables: $something
181  if (Token::level == Token::BASIC) // must at least be advanced level
182  break; // treat like ordinary char
183  if (!is.get(c)) // lookahead
184  return new Token(Token::WORD, ""); // just a $ ?
185  if (Token::level >= Token::BASH) { // bash/ksh ?
186  // The $( pair is not a variable
187  if (c == '(')
188  return new Token(Token::EVAL);
189  }
190  // The following non-letter pairs are common
191  // $* $@ $# $0 ... $9 $? $! $$
192  if (strchr("*@#0123456789?!$", c) != NULL) {
193  s += c;
194  return new Token(Token::VAR, s);
195  }
196  // Having handled all the special cases,
197  // we now expect a true word (e.g. $PATH)
198  // which MUST begin with a letter.
199  if (!isalpha(c)) {
200  is.putback(c); // Can not use this one
201  return new Token(Token::WORD, ""); // just a $ ?
202  }
203  // Oke, this looks promissing ...
204  // Collect an alfanumeric string (a _ also counts as a letter)
205  do {
206  s += c;
207  if(!is.get(c))
208  return new Token(Token::VAR, s);
209  } while (isalnum(c) || (c == '_'));
210  is.putback(c); // undo lookahead
211  return new Token(Token::VAR, s);
212 
213  case '{': // sh-script ? group begin
214  if (!(Token::level >= Token::SH))
215  break; // treat like ordinary char
216  return new Token(Token::GROUP);
217 
218  case '}': // sh-script ? group end
219  if (!(Token::level >= Token::SH))
220  break; // treat like ordinary char
221  return new Token(Token::GRPEND);
222 
223  case '`': // sh-script? (back-quotes)
224  if (!(Token::level >= Token::SH))
225  break; // treat like ordinary char
226  return new Token(Token::BQUOTE);
227 
228  } // end special character switch
229 
230  // Assume it is just a word;
231  // Read any text upto something special.
232  Token::token_t type = Token::WORD;
233 
234  do {
235  s += c;
236  if (!is.get(c))
237  return new Token(type, s);
238  if (Token::level > Token::BASIC) { // advanced ?
239  // An embedded (simple) wildcard character ?
240  // MAYBE: Handle [...] as well
241  if (strchr("*?", c) != NULL) // yes
242  type = Token::EXPR; // change type from WORD to EXPR
243  }
244  } while (!isSpecial(Token::level, c));
245  is.putback(c); // undo lookahead
246 
247  return new Token(type, s);
248 }
