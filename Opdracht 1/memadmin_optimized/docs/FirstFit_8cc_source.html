86 Area *FirstFit::searcher(int wanted)
87 {
88  require(wanted > 0); // has to be "something",

112 // Try to reclaim space by joining fragmented freespace
113 bool FirstFit::reclaim()
114 {
115  require(!areas.empty()); // sanity check
116 
117  // Sort resource map by area address
118  areas.sort(Area::orderByAddress()); // WARNING: expensive N*log(N) operation !
119 
120  // Search thru all free areas for matches between successive elements
121  bool changed = false;
122  ALiterator i = areas.begin();
123  Area *ap = *i; // The current candidate ...
124  for(++i ; i != areas.end() ;) {
125  Area *bp = *i; // ... match it with.
126  if(bp->getBase() == (ap->getBase() + ap->getSize())) {
127  // Oke; bp matches ap ... [i.e. bp follows ap]
128  ALiterator next = areas.erase(i); // remove bp from the list
129  ap->join(bp); // append area bp to ap (and destroy bp)
130  ++mergers; // update statistics
131  changed = true; // we changed something
132  i = next; // revive the 'i' iterator
133  // and now try match ap with whatever followed bp
134  } else {
135  ap = bp; // move on to next free area
136  ++i;
137  }
138  }
139  ++reclaims; // update statistics ("reclaims attempted")
140  return changed;
141 }
